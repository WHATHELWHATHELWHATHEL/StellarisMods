basic_init_galcen = {
	class = "sc_m"

	flags = { galcen galiso }

	planet = {
		count = 1
		class = star
		orbit_distance = 0
		orbit_angle = 1
		size = 30
		has_ring = no
	}

	change_orbit = 1

	planet = {
		class = "pc_habitat"
		orbit_distance = 0
		orbit_angle = 0
		size = 9
		tile_blockers = none
		modifiers = none
		init_effect = {
			# create the human species here
			create_species = {
				name = "NAME_Human"
				plural = "NAME_Human"
				class = HUM
				portrait = human
				namelist = "HUM2"
				traits = {
					trait = "trait_pc_gaia_preference"
					trait = trait_nonadaptive
					trait = trait_sedentary
					trait = trait_slow_breeders
					trait = trait_conservational
					trait = trait_venerable
				}
			}
			create_country = {
				name = "Republic of New Human"
				authority = auth_democratic
				civics = {
					civic = civic_beacon_of_liberty
					civic = civic_citizen_service
				}
				species = last_created
				ethos = {
					ethic = ethic_militarist
					ethic = ethic_fanatic_egalitarian
				}
				flag = {
					icon = {
						category = "human"
						file = "flag_human_5.dds"
					}
					background= {
						category = "backgrounds"
						file = "new_dawn.dds"
					}
					colors={
						"black"
						"blue"
						"null"
						"null"
					}
				}
				type = default
			}
			set_owner = last_created_country
			set_capital = yes
			while = {
				count = 5
				best_tile_for_pop = {
					create_pop = {
						species = last_created
					}
				}
			}
			# set center adminstration
			random_tile = {
				limit = {
					num_adjacent_tiles > 3
					has_building = no
				}
				set_deposit = d_immense_energy_deposit
				set_building = "building_hab_capital"
			}
			# create buildings each
			random_tile = {
				limit = {
					num_adjacent_tiles > 2
					num_adjacent_tiles < 4
					has_building = no
				}
				set_deposit = d_immense_farmland_deposit
				set_building = "building_hab_agri_bay"
			}
			random_tile = {
				limit = {
					num_adjacent_tiles > 2
					num_adjacent_tiles < 4
					has_building = no
				}
				set_deposit = d_immense_mineral_deposit
				set_building = "building_hab_astro_mining_facility"
			}
			random_tile = {
				limit = {
					num_adjacent_tiles > 2
					num_adjacent_tiles < 4
					has_building = no
				}
				set_deposit = d_immense_energy_deposit
				set_building = "building_hab_solar_power_processor"
			}
			random_tile = {
				limit = {
					num_adjacent_tiles > 2
					num_adjacent_tiles < 4
					has_building = no
				}
				set_building = "building_hab_laboratory_module"
			}
		}
	}

	init_effect = {
		spawn_megastructure = {
			type = ring_world_31_intermediate
			orbit_angle = 0
			orbit_distance = 40
			location = solar_system
		}
		spawn_megastructure = {
			type = ring_world_32_intermediate
			orbit_angle = 120
			orbit_distance = 40
			location = solar_system
		}
		spawn_megastructure = {
			type = ring_world_33_intermediate
			orbit_angle = 240
			orbit_distance = 40
			location = solar_system
		}
	}
}
